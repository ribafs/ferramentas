Script

Para facilitar o gerenciamento do docker

di - listar todas as imagens

docker iamges

dps - listar todos os containers, inclusive os inativos
docker ps -a

drm - listar
ocker stop $(docker ps -aq)
docker rm $(docker container ls -aq)

drmi - remover todas as images
docker rmi -f $(docker images -q)

drv - remover todos os volumes
docker volume rm $(docker volume ls -q)

dr - remover tudo
drm
drmi
drv

dcr - executar um docker-compose.yml para criar um container e o iniciar
docker-compose down -v
docker-compose up -d

Renomear todos os arquivos .md para .html de um diretório

for file in *.md; do
    mv "$file" "$(basename "$file" .md).html"
done
